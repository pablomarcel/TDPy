title: NH3H2O F-Chart/EES doc example — T=450 K, P=10 bar, X=0.5 => Hmass, Smass

given:
  # Inputs (doc values)
  T_K     = 450.0
  P_bar   = 10.0
  x_NH3   = 0.5

  # Unit conversions
  bar_to_Pa = 100000.0
  J_to_kJ   = 0.001

  # PropsSI + special NH3H2O fluid string (intercepted by coolprop_backend)
  # X is embedded in the fluid string to match your current routing approach.
  fluid = "NH3H2O|X=0.5|strict=1"

  # Reference outputs from the EES documentation example:
  h_ref_kJkg   = 2224.0
  s_ref_kJkgK  = 6.311

guess:
  # Give the solver sane initial guesses (not super critical because this is direct).
  P_Pa        = 1.0e6
  h           = 2.224e6
  s           = 6.311e3
  h_kJkg      = 2224.0
  s_kJkgK     = 6.311
  dh_kJkg     = 0.0
  ds_kJkgK    = 0.0
  h_err_pct   = 0.0
  s_err_pct   = 0.0

equations:
  # Convert pressure to SI for our PropsSI intercept path
  P_Pa = P_bar * bar_to_Pa

  # Call NH3H2O backend via PropsSI intercept (CoolProp-like outputs)
  h = PropsSI("Hmass","T",T_K,"P",P_Pa, fluid)
  s = PropsSI("Smass","T",T_K,"P",P_Pa, fluid)

  # Convert to kJ/kg and kJ/kg-K to compare to doc values
  h_kJkg  = h * J_to_kJ
  s_kJkgK = s * J_to_kJ

  # Differences vs documentation reference values
  dh_kJkg  = h_kJkg  - h_ref_kJkg
  ds_kJkgK = s_kJkgK - s_ref_kJkgK

  # Percent errors (guarded)
  h_err_pct = 100 * dh_kJkg / h_ref_kJkg
  s_err_pct = 100 * ds_kJkgK / s_ref_kJkgK

report:
  T_K, P_bar, x_NH3
  P_Pa
  h_kJkg, s_kJkgK
  h_ref_kJkg, s_ref_kJkgK
  dh_kJkg, ds_kJkgK
  h_err_pct, s_err_pct

solve:
  backend = scipy
  method = hybr
  tol = 1e-12
  max_iter = 40
