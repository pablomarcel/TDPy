  # --- dead state ---
  atm_to_Pa = 101325.0
  T0 = 25.0 + 273.15                 # K
  P0 = 1.0*atm_to_Pa                 # Pa

  # --- surroundings composition (mole fractions) ---
  y_N2_0  = 0.7686
  y_O2_0  = 0.2062
  y_Ar_0  = 0.0093
  y_CO2_0 = 0.0004
  y_H2O_0 = 0.0156

  # --- excess air ---
  ea = 0.5

  # --- methane molar mass for mass-basis exergy ---
  MW_CH4 = 16.043                    # kg/kmol

  # --- partial pressures for entropy calls (ideal-gas mixture reference) ---
  P_N2_0  = P0*y_N2_0
  P_O2_0  = P0*y_O2_0
  P_Ar_0  = P0*y_Ar_0
  P_CO2_0 = P0*y_CO2_0
  P_H2O_0 = P0*y_H2O_0

  # ================= Stoichiometric reaction coefficients =================
  # Derived from the book balances; these simplify nicely:
  # a_s = 2/y_O2_0
  a_s = 2.0/y_O2_0
  b_s = 1.0 + a_s*y_CO2_0
  c_s = 2.0 + a_s*y_H2O_0
  d_s = a_s*y_N2_0
  e_s = a_s*y_Ar_0

  # ================= Actual reaction coefficients (with excess air) =================
  a = (1.0 + ea)*a_s
  b = 1.0 + a*y_CO2_0
  c = 2.0 + a*y_H2O_0
  d = a*y_N2_0
  e_Ar = a*y_Ar_0
  f = a*y_O2_0 - 2.0

  # convenience conversions
  J_per_MJ = 1000000.0


equations:
  # --- molar specific enthalpies at dead state (J/kmol) ---
  h_CH4_0 = CTPropsSI("HMOLAR","T",T0,"P",P0,"gri30.yaml|X=CH4:1")
  h_N2_0  = CTPropsSI("HMOLAR","T",T0,"P",P0,"gri30.yaml|X=N2:1")
  h_O2_0  = CTPropsSI("HMOLAR","T",T0,"P",P0,"gri30.yaml|X=O2:1")

  h_H2O_0 = CTPropsSI("HMOLAR","T",T0,"P",P0,"gri30.yaml|X=H2O:1")

  # --- molar specific entropies at dead state (J/kmol-K) ---
  # CH4 at P0; air constituents at their partial pressures (ideal mixture convention in the book)
  s_CH4_0 = CTPropsSI("SMOLAR","T",T0,"P",P0,"gri30.yaml|X=CH4:1")
  s_N2_0  = CTPropsSI("SMOLAR","T",T0,"P",P_N2_0,"gri30.yaml|X=N2:1")
  s_O2_0  = CTPropsSI("SMOLAR","T",T0,"P",P_O2_0,"gri30.yaml|X=O2:1")
  s_Ar_0  = CTPropsSI("SMOLAR","T",T0,"P",P_Ar_0,"gri30.yaml|X=AR:1")
  s_CO2_0 = CTPropsSI("SMOLAR","T",T0,"P",P_CO2_0,"gri30.yaml|X=CO2:1")
  s_H2O_0 = CTPropsSI("SMOLAR","T",T0,"P",P_H2O_0,"gri30.yaml|X=H2O:1")

  # --- enthalpy of reactants/products per kmol CH4 ---
  H_R = h_CH4_0 + a*y_N2_0*h_N2_0 + a*y_O2_0*h_O2_0 + a*y_Ar_0*h_Ar_0 + a*y_CO2_0*h_CO2_0 + a*y_H2O_0*h_H2O_0
  H_P = b*h_CO2_0 + c*h_H2O_0 + d*h_N2_0 + e_Ar*h_Ar_0 + f*h_O2_0

  # --- entropy of reactants/products per kmol CH4 ---
  S_R = s_CH4_0 + a*y_N2_0*s_N2_0 + a*y_O2_0*s_O2_0 + a*y_Ar_0*s_Ar_0 + a*y_CO2_0*s_CO2_0 + a*y_H2O_0*s_H2O_0
  S_P = b*s_CO2_0 + c*s_H2O_0 + d*s_N2_0 + e_Ar*s_Ar_0 + f*s_O2_0

  # --- fuel exergy (J/kmol CH4) ---
  X_fuel = (H_R - H_P) - T0*(S_R - S_P)
  X_fuel_mass = X_fuel / MW_CH4

  X_fuel_MJ_per_kmol = X_fuel/J_per_MJ
  X_fuel_MJ_per_kg   = X_fuel_mass/J_per_MJ

  # --- check based on only the reacting species (CH4 + 2 O2 -> CO2 + 2 H2O) ---
  H_R_2 = h_CH4_0 + 2.0*h_O2_0
  H_P_2 = h_CO2_0 + 2.0*h_H2O_0
  S_R_2 = s_CH4_0 + 2.0*s_O2_0
  S_P_2 = s_CO2_0 + 2.0*s_H2O_0

  X_fuel_2 = (H_R_2 - H_P_2) - T0*(S_R_2 - S_P_2)
  X_fuel_2_mass = X_fuel_2 / MW_CH4
  X_fuel_2_MJ_per_kg = X_fuel_2_mass/J_per_MJ

  dX = X_fuel - X_fuel_2
  dX_mass = X_fuel_mass - X_fuel_2_mass
